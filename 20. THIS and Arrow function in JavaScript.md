```JS
const user = {
	username: "akash",
	price: 999;

	welcomeMessage: function(){
		console.log(`${this.username} , welcome to website`)
		console.log(this);
	}
}

user.welcomeMessage() // akash , welcome to website

//what if,
user.username = "sam"
user.welcomeMessage(); // Output: sam , welcome to website
```
